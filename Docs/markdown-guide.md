## Длинные блоки кода

Блоки кода автоматически получают:
- Нумерацию строк
- Возможность копирования
- Кнопку "Развернуть полностью" (для кода длиннее 15 строк)
- Полноэкранный режим
- Поиск по коду

```
import React, { useState, useEffect } from 'react';
import { motion } from 'framer-motion';

interface ComponentProps {
  title: string;
  description: string;
  isActive: boolean;
}

export const ExampleComponent: React.FC<ComponentProps> = ({
  title,
  description,
  isActive
}) => {
  const [count, setCount] = useState(0);

  useEffect(() => {
    console.log('Component mounted');
    return () => console.log('Component unmounted');
  }, []);

  const handleClick = () => {
    setCount(prev => prev + 1);
  };

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      className="p-4 rounded-lg"
    >
      <h2>{title}</h2>
      <p>{description}</p>
      <button onClick={handleClick}>
        Clicked {count} times
      </button>
    </motion.div>
  );
};
```
